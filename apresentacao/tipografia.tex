% Espaços duros
\begin{frame}[fragile]
  \frametitle{\filename{tipografia.tex}}
  \LARGE
  Espaços duros: \verb+~+\\

  % Sem espaços duros
  \only<1>{
    \begin{center}
      \fbox{
        \begin{minipage}{.5\textwidth}
          O Sr. Roberto disse que a Sra. Eduarda quer 100 páginas até o dia 5.
          Às 9 eu só havia terminado 60!
        \end{minipage}
      }
    \end{center}
  }

  % Com espaços duros
  \only<2>{
    \begin{center}
      \fbox{
        \begin{minipage}{.5\textwidth}
          O Sr.~Roberto disse que a Sra.~Eduarda quer~100 páginas até o dia~5.
          Às~9 eu só havia terminado~60!
        \end{minipage}
      }
    \end{center}
  }
\end{frame}
